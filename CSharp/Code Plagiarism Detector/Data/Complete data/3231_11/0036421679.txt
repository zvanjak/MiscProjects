#include <stdio.h>
#include <math.h>

int main () {
                int n, i;
                double red = 0.0;
                scanf("%d", &n);
                if (n<1 || n>30) printf("n nije valjan");
                else {
                     for (i=0;i<n; ++i){
                           if (i%2==0) red += pow(2,-i);
                           else             red -= pow(2,-i);
                           }
                                printf("%.12lf", red);
                }
                return 0;
}
