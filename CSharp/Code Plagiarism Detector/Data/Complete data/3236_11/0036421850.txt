#include<stdio.h>
#include<math.h>

int main() {
	int n, i;
	float x, lnx = 0;

	scanf("%d", &n);

	if (n < 1 || n > 100) {
		printf("n nije valjan");
		return 0;
	}

	scanf("%f", &x);

	if (x <= 0 || x >= 2) {
		printf("x nije valjan");
		return 0;
	}

	for (i = 1; i <= n; i++) {
		if (i % 2 == 0) {
			lnx -= pow(x-1, i)/i;
		} else {
			lnx += pow(x-1, i)/i;
		}
	}

	printf("%d %.5f %.12f", n, x, lnx);

	return 0;
}
