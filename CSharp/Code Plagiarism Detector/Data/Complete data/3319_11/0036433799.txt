   struct natjecatelj{
      int redni_br;
      char prezime[20+1];
      int bodovi;
      }bla,naj;
   naj.bodovi=0;
   int n=fread(&bla,sizeof(bla),1,f);
   while(n==1){
      if (bla.redni_br && bla.bodovi>naj.bodovi)
         naj=bla;
      n=fread(&bla,sizeof(bla),1,f);
