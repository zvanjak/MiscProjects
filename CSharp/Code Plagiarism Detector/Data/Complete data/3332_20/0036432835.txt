int NadjiNaGlavnoj(int *mat, int m, int n, int maxstup, int *max, int *min)
{
    int i,j;
    *max=*mat;
    *min=*mat;
    for(i=0; i<m; i++){
             for(j=0; j<n; j++){
                      if(i==j){
                         if (*(mat+i*maxstup+j)>*max) *max=*(mat+i*maxstup+j);
                         if (*(mat+i*maxstup+j)<*min) *min=*(mat+i*maxstup+j);
                      }
             }
    }
    return 0;
}
