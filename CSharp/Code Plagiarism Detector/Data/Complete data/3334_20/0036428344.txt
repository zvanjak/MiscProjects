
int NadjiMaxURetku(int *mat, int m, int n, int maxstup, int *retMax)
{
int i, j;
int max_red1;
int max_mat = -2147483648;
for (i=0; i<m; i++) {
max_red1 = -2147483648;
for (j=0; j<n; j++) {
if (mat[i*maxstup+j] > max_red1){
max_red1 = mat[i*maxstup+j];}
if (mat[i*maxstup+j] > max_mat){
max_mat = mat[i*maxstup+j];}
}
retMax[i] = max_red1;
}
return max_mat;
}
