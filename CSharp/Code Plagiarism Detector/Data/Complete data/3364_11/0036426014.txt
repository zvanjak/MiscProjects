#include <stdio.h>
int main() {

char a, b, c;
int a1, b1, rez;
scanf("%c %c %c", &a, &b, &c);
a1 = a - 48;
b1 = b - 48;

if (c == '+') {
	rez = a1 + b1;
	printf("%d", rez);
}
if (c == '-') {
	rez = a1 - b1;
	printf("%d", rez);
}
if (c == '*') {
	rez = a1 * b1;
	printf("%d", rez);
}
if (c == '/') {
	rez = a1 / b1;
	printf("%d", rez);
}
if (c == '%') {
	rez = a1 % b1;
	printf("%d", rez);
}
if ((c != '+') && (c != '-') && (c != '*') && (c != '/') && (c != '%')) {
	printf("Neispravna");
}
return 0;
}
