#include<stdio.h>
int main(){

  int a, b;
  int z1=10, z2=10;
  
  scanf("%d %d", &a, &b);
  
  /*Odredivanje zajednickih znamenki z1 i z2.*/
  if(a%10 == b%10 || a%10 == b/10)
    z1 = a%10;
    
  if(a/10 == b%10 || a/10 == b/10)
    z2 = a/10;
    
  /*Ako su jednaki pocetnoj vrijednosti, nema presjeka.*/
  if(z1==10 & z2==10){
    printf("Nema");
  }
  /*Ako su z1 i z2 razliciti, ali je jedan od njih jo≈° uvijek u pocetnoj vrijednosti.*/
  else if( (z1!=10 || z2!=10) && (z1==10 || z2==10) ){  
    printf("%d", z1==10 ? z2 : z1);
  }
  /*Ako oba nisu u pocetnoj vrijednosti (Postoje dvije zajednicke znamenke).*/
  else{
    if(z1<z2)
      printf("%d %d", z1, z2);
    else if(z2<z1)
      printf("%d %d", z2, z1);
    else
      printf("%d", z1);
  }
  
  return 0;
}
