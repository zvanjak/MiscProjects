int *napraviPolje(int *a,int na,int *b,int nb,int *nc){
    int i;
    int *c;
    c=(int *)malloc((na+nb)*4);
    *nc=0;
    for(i=0;i<na;i++)
       if ((a[i]>9 && a[i]<100) || (a[i]<-9 && a[i]>-100))
          c[(*nc)++]=a[i];
    for(i=0;i<nb;i++)
       if ((b[i]>9 && b[i]<100) || (b[i]<-9 && b[i]>-100))
          c[(*nc)++]=b[i];
    c=(int *)realloc(c,(*nc)*4);
    return c;
    }
