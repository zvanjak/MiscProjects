#include <stdio.h>
#include <malloc.h>

int *napraviPolje(int *A,int Na, int *B,int Nb,int *Nc){
int i,j,f=0,*p,k=0;
if(Na==0 || Nb==0) return NULL;
for (i=0;i<Na;i++){
k=0;
for (j=0;j<Nb;j++){
if (A[i] > B[j]) k++;
}
if (k==Nb) f++;
}
p = (int *) malloc (f*sizeof (int));
f=0;
for (i=0;i<Na;i++){
k=0;
for (j=0;j<Nb;j++){
if (A[i] > B[j]) k++;
}
if (k==Nb) {
*(p+f)=A[i];
f++;
}
}
*Nc=f;
return p;
}
