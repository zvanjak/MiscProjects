int *napraviPolje(int *A, int Na, int *B, int Nb, int *Nc) {
	int i, j, min;
	int *C;

	*Nc=0;
	min=B[0];
	for(i=1; i<Nb; i++) {
		if(B[i]<min) min=B[i];
	}

	for (i=0; i<Na; i++) {
		if(A[i]<min) (*Nc)++;
	}

	C=(int *)malloc(*Nc * sizeof(int));
	
	j=0;
	for (i=0; i<Na; i++) {
		if(A[i]<min) C[j++]=A[i];
	}

	return C;
}
