int* napraviPolje(int *A, int Na, int *B, int Nb, int *Nc)
{
    int i, min;
    int *C;
    C = (int *)malloc(Na * sizeof(int));
    
    min = B[0];
    for(i = 0; i < Nb; i++)
        if(B[i] < min)
            min = B[i];
            
    *Nc = 0;
    for(i = 0; i < Na; i++)
        if(A[i] < min)
            C[(*Nc)++] = A[i];
          
    C = (int *)realloc(C, (*Nc) * sizeof(int));
    
    return(C);
}
