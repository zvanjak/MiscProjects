#include <stdio.h>

void main () {
	int i, j, n, x=0, o=0, a=0, b=0;
	char mat[20][20];
	scanf("%d", &n);
	for(i=0; i<n; i++) {
		for(j=0; j<n; j++) {
			scanf("%c", &mat[i][j]);
		}
	}
	for(i=0; i<n; i++) {
		for(j=0; j<n; j++) {
			if(mat[i][j]=='X') a++;
			if(mat[i][j]=='O') b++;
		}
		if(a==n) x=1;
		if(b==n) o=1;
		a=0;
		b=0;
	}
	for(i=0; i<n; i++) {
		for(j=0; j<n; j++) {
			if(mat[j][i]=='X') a++;
			if(mat[j][i]=='O') b++;
				}
		if(a==n) x=1;
		if(b==n) o=1;
		a=0;
		b=0;
	}
	for(i=0; i<n; i++) {
		for(j=0; j<n; j++) {
			if(i==j){
				if(mat[i][j]=='X') a++;
				if(mat[i][j]=='O') b++;
			}
		}
	}
	if(a==n) x=1;
	if(b==n) o=1;
	a=0;
	b=0;
	for(i=0; i<n; i++) {
		if(mat[i][n-i-1]=='X') a++;
		if(mat[i][n-i-1]=='O') b++;
	}
	if(a==n) x=1;
	if(b==n) o=1;
	if((x==1 && o==1) || (x==0 && o==0)) printf("-");
	if(x==1 && o==0) printf("X");
	if(x==0 && o==1) printf("O");
}
