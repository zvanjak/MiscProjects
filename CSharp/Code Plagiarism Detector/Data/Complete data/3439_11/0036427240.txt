#include <stdio.h>
int main(){
	int matrica[20][20], n, greska=0, produkt=1, i, j;
	scanf("%d", &n);
	for (i=0; i<n; i++)
		for (j=0; j<n; j++){
			scanf ("%d", &matrica[i][j]);
			if ((j%2==0)&&(i%2==0)) if (matrica[i][j]%2!=0) greska=1;
		}
	if (!greska)
		for (i=0; i<n; i++)
			produkt*=matrica[i][i];
	else
		for (i=0; i<n; i++)
				produkt*=matrica[i][n-1-i];
	printf("%d", produkt);
	return 0;
}
