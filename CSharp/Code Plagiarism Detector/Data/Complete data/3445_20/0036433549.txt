class Kosarica {
public:
          Kosarica() {
                 suma=0;
	     max=0; }
	void DodajProizvod(float cijena) {
	    suma+=cijena;
	    if (cijena>max) max=cijena;}
	float UkupnaCijena() {return suma;}
	float NajskupljiProizvod() {return max;}
        ~Kosarica() {}
private:
	float suma;
	float max;
};
