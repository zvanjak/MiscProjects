class Kosarica{
float _cijena;
float _najcijena;
float _suma;
float _brojac;
public:
Kosarica()
{
_cijena=0;
_najcijena=0;
_suma=0;
_brojac=0;
}
void DodajProizvod(float cijena)
{
_cijena=cijena;
if(_brojac==0) _najcijena=cijena;
if(_najcijena>cijena)
_najcijena=cijena;
_suma=_suma+cijena;
_brojac=_brojac+1;
}
float ProsjecnaCijena()
{
if(_brojac==0) return 0;
return _suma/_brojac;
}
float NajjeftinijiProizvod()
{
if(_brojac==0) return 0;
return _najcijena;
}
~Kosarica(){};
}
