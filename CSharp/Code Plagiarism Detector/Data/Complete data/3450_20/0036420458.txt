class Predstava
{
private:
     float _pocetak;
     float _kraj;
     int _maxBrojKarata;
     int _prodanoKarata;
public:
     Predstava();
     Predstava(float pocetak, float kraj, int maxBrojKarata);
     float Trajanje();
     void ProdajKartu();
     int ProdanoKarata();
     int PreostaloKarata();
};
Predstava::Predstava(){
         _pocetak = 8.0;
         _kraj = 10.0;
         _maxBrojKarata = 100; 
         _prodanoKarata = 0;
}
Predstava::Predstava(float pocetak, float kraj, int maxBrojKarata){
         _pocetak = pocetak;
         _kraj = kraj;
         _maxBrojKarata = maxBrojKarata; 
         _prodanoKarata = 0;
}
float Predstava::Trajanje(){
         return _kraj - _pocetak;
}
void Predstava::ProdajKartu(){
         if(_prodanoKarata < _maxBrojKarata) _prodanoKarata = _prodanoKarata + 1;
}
int Predstava::ProdanoKarata(){
         return _prodanoKarata;
}
int Predstava::PreostaloKarata(){
         return _maxBrojKarata - _prodanoKarata;
}
