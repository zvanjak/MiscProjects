int LijevoDesno(cvor *glava,int id,int *lijevo,int *desno){
 if (glava==0) return 0;
 if(id<glava->id){   (*lijevo)++;
                      return LijevoDesno(glava->lijevo,id,lijevo,desno);
                 }
 else if(id>glava->id){ (*desno)++;
                        return LijevoDesno(glava->desno,id,lijevo,desno);
                      }
 else return 1;
}
