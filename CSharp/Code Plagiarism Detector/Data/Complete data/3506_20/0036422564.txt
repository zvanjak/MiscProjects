int Razina(cvor *glava, int id, int *razina){
    if (glava==NULL) {
       if ((*razina)==1) (*razina)=0;
       return 0; 
    };
      
    if (glava->id==id) return 1;   
    (*razina)++; 
    if (Razina(glava->lijevo,id,razina)) return 1;    
    if (Razina(glava->desno,id,razina)) return 1;
    (*razina)--;
    return 0;
}
