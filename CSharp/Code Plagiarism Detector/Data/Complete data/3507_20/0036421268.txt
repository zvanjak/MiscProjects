int Sortirano (cvor *glava){
	int i, j;
	i=j=0; 
	
	if (glava == NULL)
		return 1;
	else {
		if (glava->lijevo == NULL)
			i=1;
		else if (glava->lijevo->id < glava->id)
			i = Sortirano (glava->lijevo);
		if (glava->desno == NULL)
			j=1;
		else if (glava->desno->id >= glava->id)
			j = Sortirano (glava->desno);
	}
	return (i && j);
}
