int Sortirano (cvor *glava)
{
if (!glava)
return 1;
int l = 0, r = 0;
int ls = 0, rs = 0; 
ls = Sortirano (glava->lijevo);
if (glava->lijevo)
{
if (glava->id > glava->lijevo->id)
l = 1;
}
else
l = 1;
rs = Sortirano (glava->desno);
if (glava->desno)
{
if (glava->id < glava->desno->id)
r = 1;
}
else
r = 1;
return l & ls & r & rs;}
