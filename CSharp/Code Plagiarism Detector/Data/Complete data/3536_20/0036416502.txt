void UmetniPrijePrvogNeparnog(Stog * stog, int broj) {
	Stog pomStog;
	int vrh;
	int nasao;

	if (!stog->Skini(&vrh))
		stog->Stavi(broj);
	else {
		stog->Stavi(vrh);
		nasao = 0;
		while (stog->Skini(&vrh)) {
			if (vrh % 2 == 1) {
				stog->Stavi(vrh);
				stog->Stavi(broj);
				nasao = 1;
				break;
			}
			else {
				pomStog.Stavi(vrh);
			}
		}
		
		if (!nasao)
			stog->Stavi(broj);
		while (pomStog.Skini(&vrh)) {
			stog->Stavi(vrh);
		}
	}
}
