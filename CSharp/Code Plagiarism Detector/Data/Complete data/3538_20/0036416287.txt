Red *VratiBrojeve(Red *red){
Red pom;
Red *red2=new Red;
int var,tmp,pot;
while(red->SkiniIzReda(&var)){
pom.DodajURed(var);
tmp=0;
for(int i=var-1;i>=0;i--){
red->SkiniIzReda(&var);
pom.DodajURed(var);
pot=1;
for(int t=i;t>0;t--) pot*=10;
tmp+=var*pot;
}
red2->DodajURed(tmp);
}
while(pom.SkiniIzReda(&var)) red->DodajURed(var);
return red2;
}
