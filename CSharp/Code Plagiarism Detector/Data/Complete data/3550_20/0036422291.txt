int izbacinaj (zapis **glava) {
    zapis *pom, *i, *min, *max;
    int n = 0;
    float prosjek, suma = 0;

    if (*glava == NULL) return 0;
    for (i = *glava; i; i = i->sljed) {
        n++;
        suma += i->pros;
    }
    if (n == 1) { 
        pom = *glava;
        *glava = NULL;
        free (pom);
        return 1;
    }
    prosjek = suma / n;
    min = max = *glava;
    for (i = *glava; i->sljed; i = i->sljed) {
        if ((i->sljed)->pros < min->pros) min = i->sljed;
        else if ((i->sljed)->pros > max->pros) max = i->sljed;
    }
    if (min->pros == max->pros) {
        pom = *glava;
        *glava = (*glava)->sljed;
        free (pom);
        pom = *glava;
        if ((*glava)->sljed) {
            *glava = (*glava)->sljed;
            free (pom);
        }
        else {
            *glava = NULL;
            free (pom);
        }
        return 1;
    }
    else {
        if ((i = *glava)->matbr != max->matbr) {
            for (i = *glava; i->sljed && (i->sljed)->matbr != max->matbr; i = i->sljed);
            pom = i->sljed;
            i->sljed = pom->sljed;
        }
        else {
            pom = *glava;
            *glava = (*glava)->sljed;
        }
        free (pom);
        if ((i = *glava)->matbr != min->matbr) {
            for (i = *glava; i->sljed && (i->sljed)->matbr != min->matbr; i = i->sljed);
            pom = i->sljed;
            i->sljed = pom->sljed;
        }
        else {
            pom = *glava;
            *glava = (*glava)->sljed;
        }
        free (pom);
        return 2;
    }
}
