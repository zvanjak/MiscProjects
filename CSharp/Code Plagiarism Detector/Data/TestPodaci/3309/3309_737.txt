int br_prodanih(FILE *f, char *autor){
	struct album{
		char naziv[30+1];
		char autor[20+1];
		int prodano;
	}zapis;
	int prodano = 0;
	fseek (f, (long) 0, SEEK_SET );
	while(fread(&zapis, sizeof(struct album), 1, f) != 0){
		if (strcmp(zapis.autor, autor) == 0)
			prodano += zapis.prodano;
	}
	return prodano;
}
