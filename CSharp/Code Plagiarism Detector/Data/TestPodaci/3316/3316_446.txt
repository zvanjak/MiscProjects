void azuriraj( FILE *f, int redni_br, int bodovi ) {
   struct natjecatelj A;

   fseek( f, (redni_br-1)*sizeof(struct natjecatelj), SEEK_SET );
   fread( &A, sizeof(struct natjecatelj), 1, f );

   A.bodovi = bodovi;

   fseek( f, (redni_br-1)*sizeof(struct natjecatelj), SEEK_SET );
   fwrite( &A, sizeof(struct natjecatelj), 1, f );
}
