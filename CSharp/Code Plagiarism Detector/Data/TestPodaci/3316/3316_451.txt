void azuriraj(FILE *f, int redni_br, int bodovi)
{
    struct natjecatelj
    {
    int redni_br;
    char prezime[20+1];
    int bodovi;
    } zapis; 
    
    fseek(f, 0L, SEEK_SET);
    while(fread(&zapis, sizeof(zapis), 1, f) == 1)
    {
        if(zapis.redni_br == redni_br)
        {
            zapis.bodovi = bodovi;
            break;
        }
    }
    
    fseek(f, -1L * sizeof(zapis), SEEK_CUR);
    fwrite(&zapis, sizeof(zapis), 1, f);    
}
