#include <stdio.h>


int main(){
int x[100],i,n,b,brop=0,br=0;

do{
scanf("%d",&n);
}while(n<1 || n>100); 

for(i=0;i<n;i++){
scanf("%d",&x[i]);
}


do{
scanf("%d %d",&i,&b);
if(i>(n-1) || i<0) break;
else {
x[i]/=b;
brop++;
}

}while(1);

for(i=0;i<n;i++)
if(x[i]==0) br++;

printf("%d %d",brop,br);

return 0;
}
