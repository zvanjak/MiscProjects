#include<stdio.h>
int main () {
	char a[100];
	int i, b;
	b=0;
	for(i=0; i<100; i++) {
		scanf("%c", &a[i]);
		b++;
		if (a[i]=='.') i=99;
	}
	for (i=0; i<b; i++) {
		if ((a[i]>64)&&(a[i]<91)) a[i]+=32;
	}
	if ((a[0]>96)&&(a[0]<123)) a[0]-=32;
	for (i=0; i<b; i++) {
		if (a[i]==32) {
			if ((a[i+1]>96)&&(a[i+1]<123)) a[i+1]-=32;
			if ((a[i-1]>96)&&(a[i-1]<123)) a[i-1]-=32;
			else if (a[i-1]==',') {
				if ((a[i-2]>96)&&(a[i-2]<123)) a[i-2]-=32;
			}

		}
	}
	if ((a[b-2]>96)&&(a[b-2]<123)) a[b-2]-=32;
	for (i=0; i<b; i++) {
		printf("%c", a[i]);
	}
	return 0;
}
