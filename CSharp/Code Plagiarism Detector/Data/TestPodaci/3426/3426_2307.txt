#include <stdio.h>
#define maxmr 20
#define maxms 5
int mat[maxmr][maxms];
int mr,ms,p,q,i,j,k,g,br_zamjena,br_s_bez_nule,kontrola;

int main()
{

do
{
scanf("%d %d",&mr,&ms);
} while (mr<1||mr>maxmr||ms<1||ms>maxms);
scanf("%d %d",&p,&q);

for (k=0;k<mr;k++)
{


for (g=0;g<ms;g++)
{
if (k%2==0) mat[k][g]=p;
if (k%2==1) mat[k][g]=q;
}

}

do
{
scanf("%d %d",&i,&j);

if ((i>=0&&i<mr)&&(j>=0&&j<ms)) 
{
mat[i][j]=0;
br_zamjena++;
}

} while ((i>=0&&i<mr)&&(j>=0&&j<ms));

br_s_bez_nule=ms;
for (k=0;k<ms;k++)
{ kontrola=0;
	for (g=0;g<mr;g++)
	{
	if (mat[g][k]==0) kontrola=1;
	}

	if (kontrola) br_s_bez_nule--;
}

printf("%d %d", br_zamjena,br_s_bez_nule);

return 0;

}
