#include <stdio.h>

int main () {
	char znamenka;
	int i = 0, sumap = 0, suman = 0;
	do {
		scanf("%c", &znamenka);
if (znamenka == 10) scanf("%c", &znamenka);
		if ((znamenka >= '0') && (znamenka <= '9')) {
			if (znamenka % 2) {
				suman += znamenka - '0';
			}else {
				sumap += znamenka - '0';
			}
		}
		else if ((znamenka >= 'A') && (znamenka <= 'F')) {
			if (znamenka % 2) {
				sumap += znamenka - 'A' + 10;
			}else {
				suman += znamenka - 'A' + 10;
			}
		}
		else if ((znamenka >= 'a') && (znamenka <= 'f')) {
			if (znamenka % 2) {
				sumap += znamenka - 'a' + 10;
			}else {
				suman += znamenka - 'a' + 10;
			}
		}
		else break;
		i++;
	} while (i<100);
	printf("%d", sumap-suman);
return 0;
}
