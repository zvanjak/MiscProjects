#include <stdio.h>
#define MAX 100

int main () {
          int a[MAX], i, j, brojac = 0, b;
          for(i=0; i<MAX; i++) {
                   scanf("%d", &a[i]);
                   if(a[i]==-1) break;
                   brojac++;
          }
          for(i=0; i<brojac; i++){
                   if(i%2==1) {
                           b=a[i];
                           for(j=0; j<b; j++) {
                                    printf("%d ", a[i-1]);
                           }
                   }
          }
return 0;
}
